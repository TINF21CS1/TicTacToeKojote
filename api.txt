lobby/join
    -> player-profile
    <- statistics
    b- current players, player status

lobby/kick
    -> player-profile
    b- player kicked

lobby/ready
    -> own ready status
    b- ready status of all players
    ( b- all ready, game starting )

game/make_move
    -> coordinates
    b- gamestate, active_player || ( <- errormessage ) || b- game finished, (winner)

chat/message
    -> text
    b- player, text

[[Player Leaves Game (Socket connClose)]]
    b- player left game, game finished, winner
